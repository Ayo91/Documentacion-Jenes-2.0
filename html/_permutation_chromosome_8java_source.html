118  public final void cross(final PermutationChromosome chromosome, final int from, int to) {
119 
120  // This operation is possible only if both chromosomes entail
121  // the same number of permutation elements
122  if (chromosome.permutation.length != this.permutation.length) {
123  throw new RuntimeException("Incompatible chromosome length.");
124  }
125 
126  to += 1;
127  if (to > this.permutation.length) {
128  to = this.permutation.length;
129  }
130 
131  // The original values in the crossover area are
132  // saved in the temporary structures
133  for (int i = from; i < to; ++i) {
134  this.temp_permutation[i] = this.permutation[i];
135  chromosome.temp_permutation[i] = chromosome.permutation[i];
136  }
137 
138  // The elements out the crossover area are left unchanged
139  for (int i = from; i < to; ++i) {
140 
141  // This is the element of this chromosome at position i
142  // within the crossover area
143  int this_elem = this.temp_permutation[i];
144 
145  // We look for the position of this element in the other chromosome
146  int chrm_pos = chromosome.position[this_elem];
147 
148  // If this_elem is also in the crossover area for the other chromosome
149  if (chrm_pos >= from && chrm_pos < to) {
150  // We take the element in position i, as provided
151  // by this chromosome schema. The position is exchanged with
152  // the element that currently occupies that position
153  chromosome.swap(chrm_pos, i);
154  }
155 
156  // In dual way, we apply the same strategy to the other chromosome.
157  // This is the element of the other chromosome in position i
158  int chrm_elem = chromosome.temp_permutation[i];
159 
160  // And this is the position held by this element in this chromosome.
161  int this_pos = this.position[chrm_elem];
162 
163  // If it is within the crossover area
164  if (this_pos >= from && chrm_pos < to) {
165  // We take this element in position i
166  this.swap(this_pos, i);
167  }
168  }
169  }
