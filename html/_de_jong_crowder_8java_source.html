235  protected void replace(Population<T> initial, Population<T> preselected, Population<T> evolved, Population<T> out) {
236 
237  out.setAs(initial);
238 
239  int len = out.size();
240 
241  double[] diff = null;
242 
243  for (Individual<T> offspring : evolved) {
244 
245  Individual<T> r = null;
246  double s = Double.NEGATIVE_INFINITY;
247 
248  for (int i = 0; i < crowdingFactor; ++i) {
249  int k = this.random.nextInt(len);
250  Individual<T> ind = out.getIndividual(k);
251 
252  diff = Chromosome.Util.getDifference(offspring.getChromosome(), ind.getChromosome(), diff);
253  double sm = similarity(diff);
254 
255  if (sm > s) {
256  r = ind;
257  s = sm;
258  }
259  }
260 
261  if (elitist) {
262 
263  Fitness fit = this.getFitness();
264 
265  boolean better = fit != null ? fit.dominates(offspring, r) : Fitness.dominates(offspring, r, this.biggerIsBetter);
266 
267  if (better) {
268  r.setAs(offspring);
269  }
270  } else {
271  r.setAs(offspring);
272  }
273 
274  }
275  }
