79  @Override
80  public void evaluateIndividual(Individual<BitwiseChromosome> individual) {
81  BitwiseChromosome chrom = individual.getChromosome();
82  double score = 0.0;
83 
85  int foundFirstLevelBlocks = 0;
86  for (int block = 0; block < NBLOCKS; block++) { // for each first level block
87 
88  int num_bits = 0;
89 
90  int block_start = block * BLOCK_DELTA;
91  int next_block = block_start + BLOCKSIZE;
92 
93  // count the bits in the current block
94  for (int j = block_start; j < next_block; j++) {
95  if (chrom.getBitValueAt(j) == 1) {
96  num_bits++;
97  }
98  }
99 
100  // assign the score according to the number of bits found
101  if (num_bits > BLOCK_THRESHOLD && num_bits < BLOCKSIZE) {
102  score -= ((num_bits - BLOCK_THRESHOLD) * PART_CALCULATION_STEP);
103  } else if (num_bits <= BLOCK_THRESHOLD) {
104  score += (num_bits * PART_CALCULATION_STEP);
105  }
106 
107  // mark which lowest level blocks were found
108  if (num_bits == BLOCKSIZE) {
109  block_array[block] = 1;
110  foundFirstLevelBlocks++;
111  } else {
112  block_array[block] = 0;
113  }
114  }
115 
116  // assign bonus for filled first level blocks
117  if (foundFirstLevelBlocks > 0) {
118  score += (FIRST_BONUS + ((foundFirstLevelBlocks - 1) * FOLLOWING_BONUS));
119  }
120 
121 
123  // max blocks in current level
124  int mx_blocks = NBLOCKS; // now number of filled low level blocks
125 
126  boolean proceed = true; // should we look at the next higher level?
127  int level = 0;
128 
129  while ((mx_blocks > 1) && proceed) {
130  proceed = false;
131 
132  int blocks_found = 0; //number of blocks found with size equal to mx_blocks
133 
134  // there are mx_blocks valid blocks in the blockarray each time
135  // round, so mx_blocks=2 is the last.
136 
137  for (int block = 0, index = 0; block < (mx_blocks / 2) * 2; block += 2, index++) {
138  if (block_array[block] == 1 && block_array[block + 1] == 1) {
139  blocks_found++;
140  block_array[index] = 1; //there is a block with size equal to mx_block
141  proceed = true;
142  } else {
143  block_array[index] = 0;
144  }
145  }
146  if (blocks_found > 0) {
147  score += FIRST_BONUS + (blocks_found - 1) * FOLLOWING_BONUS;
148  level++;
149  }
150  //max blocks in next level decrease
151  mx_blocks /= 2;
152  }
153 
154  individual.setScore(score);
155  }
