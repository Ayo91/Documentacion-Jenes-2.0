245  public SimpleGA(final Fitness<T> fitness, final Population<T> population, final int generations, final double crossover, final double mutation, final int elitism, final SelectionMethod selmethod, final CrossoverMethod crossmethod, ElitismStrategy es) {
246  super(fitness, population, generations);
247 
248  selector =
249  selmethod == SelectionMethod.TOURNAMENT ? new TournamentSelector<T>(2) : new RouletteWheelSelector<T>();
250 
251  crossover_op =
252  crossmethod == CrossoverMethod.SINGLEPOINT ? new OnePointCrossover<T>(crossover) : new TwoPointsCrossover<T>(crossover);
253 
254  mutator = new SimpleMutator<T>(mutation);
255 
256  this.elitism = elitism;
257  this.addStage(selector);
258  this.addStage(crossover_op);
259  this.addStage(mutator);
260  }
