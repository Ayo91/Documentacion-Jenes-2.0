154  private void map(String[] schema) {
155  for (String key : schema) {
156  for (WritableSheet sh : workbook.getSheets()) {
157  Cell cell = sh.findCell(key, 0, 0, sh.getColumns() - 1, 0, false);
158  if (cell != null) {
159  int col = cell.getColumn();
160  this.sheets.put(key, sh);
161  this.putInColumns(sh, key, col);
162  this.alterableSchema.put(key, true);
163  break;
164  } else if (key.equalsIgnoreCase(sh.getName())) {
165  /*if the cell does not exist, it is checked if key matches the name of the sheet */
166  this.sheets.put(key, sh);
167  //create new columns
168  HashMap<String, Integer> cols = new HashMap<String, Integer>();
169  cols.put(key, 0);
170  this.colsOfSheet.put(sh, cols);
171  this.alterableSchema.put(key, true);
172  }
173  }
174  }

303  private WritableCell fillCell(Object value, String key, int currentLine, int c, WritableSheet sh, boolean valueAsArray) throws WriteException {
304  WritableCell cell = null;
305 
306  if (value instanceof java.lang.Double) {
307  double d = (Double) value;
308  cell = new Number(c, currentLine, d);
309  } else if (value instanceof java.lang.Float) {
310  float f = (Float) value;
311  cell = new Number(c, currentLine, f);
312  } else if (value instanceof java.lang.Integer) {
313  int i = (Integer) value;
314  cell = new Number(c, currentLine, i);
315  } else if (value instanceof java.lang.Long) {
316  long l = (Long) value;
317  cell = new Number(c, currentLine, l);
318  } else if (value instanceof java.util.Date) {
319  Date d = (Date) value;
320  cell = new DateTime(c, currentLine, d);
321  } else if (value.getClass().isArray()) {
322  //if we are storing an array
323  Object[] arr = this.castArray(value);
324  //alterableSchema tells if a column could be altered (true only the first time)
325  if (this.alterableSchema.get(key) && arr.length > 1) {
326  this.alterSchema(sh, c + 1, arr.length);
327  this.alterableSchema.put(key, false); //reset alterable flag
328  }
329 
330  if (valueAsArray) {
331  //second or greater level of nesting in array
332  cell = new Label(c, currentLine, value.toString());
333  } else {
334  //first level of nesting for array
335  for (Object obj : arr) {
336  //recursive call -> we store a single cell of the array (the alter is already performed)
337  this.newCols.add(this.fillCell(obj, key, currentLine, c, sh, true));
338  //...change column...
339  c++;
340  }
341  }
342  } else if (value instanceof Chromosome) {
343  Chromosome chr = (Chromosome) value;
344  for (Object obj : chr.toArray()) { //XXX check in the case of bit array
345  //recursive call -> store each allele
346  this.newCols.add(this.fillCell(obj, key, currentLine, c, sh, true));
347  //change column
348  c++;
349  }
350 
351  } else if (value instanceof Individual) {
352  //in the case of Individual we alter the Sheet
353  Individual ind = (Individual) value;
354 
355  if (this.alterableSchema.get(key)) {
356  this.alterSchema(sh, c + 1, ind.getChromosomeLength() + ind.getNumOfObjectives());
357  this.alterableSchema.put(key, false);
358  }
359 
360  for (Object obj : ind.getChromosome().toArray()) { //XXX check in the case of bit array
361  //recursive call -> store each allele
362  this.newCols.add(this.fillCell(obj, key, currentLine, c, sh, true));
363  //change column
364  c++;
365  }
366 
367  for (Object obj : ind.getAllScores()) {
368  //recursive call -> store scores
369  this.newCols.add(this.fillCell(obj, key, currentLine, c, sh, true));
370  //change column
371  c++;
372  }
373 
374 
375  } else if (value instanceof Group) {
376 
377  //in the case of Group we create a new dedicated Sheet
378  Group<Chromosome> grp = (Group<Chromosome>) value;
379  int groupSaparator = 1;
380  if (this.alterableSchema.get(key)) {
381  Individual<Chromosome> individual = grp.get(0);
382  if (individual != null) {
383  WritableCell[] cells = new WritableCell[2];
384 
385  cells[0] = new Label(0, 0, CHROMOSOME_LABEL);
386  cells[1] = new Label(1, 0, SCORES_LABEL);
387 
388  int len = individual.getChromosomeLength();
389 
390  WritableSheet otherSheet = this.editOtherSheet(key, individual.getChromosomeLength(), cells);
391  otherSheet.mergeCells(len, 0, len + individual.getAllScores().length - 1, 0);
392  groupSaparator = 0;
393  //remove column from first sheet
394  if (!otherSheet.equals(sh)) {
395  //adjust schema in main sheet removing the column indicating group name
396  this.removeColumnFromFirstSheet(key, sh);
397  }
398  }
399  this.alterableSchema.put(key, false); //reset flag
400  }
401 
402  //access new sheet just created
403  WritableSheet ws = this.sheets.get(key);
404  int curLine = ws.getRows() + groupSaparator;
405  for (Object ind : grp) {
406  //... and fill it appending rows
407  this.fillCell((Individual) ind, key, curLine, 0, ws, true);
408 
409  curLine++;
410  }
411  } else {
412  //... unknown type to store in logger
413  String s = value.toString();
414  cell = new Label(c, currentLine, s);
415  }
416  return cell;
417  }

425  private void alterSchema(WritableSheet sh, int startIndex, int length) {
426  Range[] mergedCells = sh.getMergedCells();
427 
428  int cellMerged = 0;
429  for (Range range : mergedCells) {
430  //... first cell at left
431  Cell topLeft = range.getTopLeft();
432 
433  //... check merged only if we are in the first row
434  if (topLeft.getRow() == 0 && topLeft.getColumn() == startIndex - 1) {
435  cellMerged = range.getBottomRight().getColumn() - topLeft.getColumn();
436  length = length - cellMerged;
437 
438  break;
439  }
440  }
441 
442  if (length > 1) { //only in the case of aumented dimension
443  for (int c = 1; c < length; c++) {
444  sh.insertColumn(startIndex);
445  }
446 
447  try {
448  sh.mergeCells(startIndex - 1, 0, startIndex - 1 + length - 1, 0);
449  } catch (WriteException ex) {
450  Logger.getLogger(XLSLogger.class.getName()).log(Level.SEVERE, null, ex);
451  }
452  }
453 
454  //... update references in sheets
455  Map<String, Integer> cols = this.colsOfSheet.get(sh);
456  Map<String, Integer> temp = new HashMap<String, Integer>(cols);
457  for (String key : temp.keySet()) {
458  int i = temp.get(key);
459  if (i >= startIndex) {
460  cols.put(key, i + length - 1);
461  }
462  }
463  }

489  private WritableSheet editOtherSheet(String key, int columns, WritableCell... cells) throws WriteException {
490  //create a new sheet
491  WritableSheet sheet = this.workbook.getSheet(key);
492  if (sheet == null) {
493  sheet = this.workbook.createSheet(key, this.workbook.getNumberOfSheets() + 1);
494  for (WritableCell cell : cells) {
495  sheet.addCell(cell);
496  }
497  this.sheets.put(key, sheet);
498  //create new columns
499  HashMap<String, Integer> cols = new HashMap<String, Integer>();
500  cols.put(key, 0);
501  this.colsOfSheet.put(sheet, cols);
502  }
503  //edit new sheet schema
504  this.alterSchema(sheet, 1, columns);
505  return sheet;
506  }

513  private void removeColumnFromFirstSheet(String key, WritableSheet sh) {
514  Map<String, Integer> cols = this.colsOfSheet.get(sh);
515 
516  int index = cols.get(key);
517  cols.remove(key);
518  Map<String, Integer> temp = new HashMap<String, Integer>(cols);
519 
520  //update column index and reference before remove from the sheet
521  for (String colName : temp.keySet()) {
522  int colNumber = temp.get(colName);
523  if (colNumber > index) {
524  cols.put(colName, --colNumber);
525  }
526  }
527  sh.removeColumn(index);
528  }
