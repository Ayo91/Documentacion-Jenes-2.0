254  public final Object getValueAt(final int index) {
255  if (index < 0 || index >= size) {
256  throw new ArrayIndexOutOfBoundsException();
257  }
258 
259  if (coding != null) {
260  int k = index / slots;
261  int h = index % slots;
262 
263  // WARNING: The following solution does not preserve the sign.
264  // For example if we consider the chromosome xx xx F1 xx
265  // this becomes 00 00 00 F1 that is positive. Instead it should be FF FF FF F1
266  // int v = (genes[k] >> (slots - h -1)*coding.SIZE.BITS) & coding.SIZE.MASK;
267 
268  // The following solution preserves the sign.
269  // Indeed the first line moves the value at the head: xx xx F1 xx becomes F1 00 00 00
270  // The second line moves the head value to the bottom preserving the sign: F1 00 00 00 -> FF FF FF F1
271  int v = genes[k] << h * coding.SIZE.BITS;
272  v >>= (slots - 1) * coding.SIZE.BITS;
273  return coding.decode(v);
274  } else {
275  return genes[index];
276  }
277  }

287  if (index < 0 || index >= size) {
288  throw new ArrayIndexOutOfBoundsException();
289  }
290 
291  if (coding != null) {
292  int k = index / slots;
293  int h = index % slots;
294  int v = coding.encode(value);
295  int mask = coding.SIZE.MASK << ((slots - 1 - h) * coding.SIZE.BITS);
296  genes[k] &= ~mask;
297  v = v << ((slots - 1 - h) * coding.SIZE.BITS);
298  genes[k] |= (mask & v);

440  private final void bitcross(final int offset, final int index, final BitwiseChromosome chromosome, final boolean after) {
441 
442  // If offset is 0 and the crossover is on left,
443  // or the offset is the last bit and the crossover is on right
444  // there is nothing to cross;
445  if (offset == 0 && !after || offset == bits - 1 && after) {
446  return;
447  }
448 
449  int mask = 0xFFFFFFFF >>> (Integer.SIZE - offset);
450 
451  if (!after) {
452  mask = ~mask;
453  }
454 
455  int v_this = this.genes[index];
456  int v_chrom = chromosome.genes[index];
457 
458  this.genes[index] = (v_this & ~mask) | (v_chrom & mask);
459  chromosome.genes[index] = (v_chrom & ~mask) | (v_this & mask);
460  }

597  private final int bitshift(final int index, final int from, final int to, final int bit) {
598  if (from > to) {
599  // right shift
600  int submask_1 = to == 0 ? 0 : (0xFFFFFFFF >>> (bits - to));
601  int submask_2 = (from + 1) == bits ? 0 : (0xFFFFFFFF << (from + 1));
602  int mask = ~(submask_1 ^ submask_2);
603 
604  int value = this.genes[index] & mask;
605  int b = (this.genes[index] >>> (to)) & 1;
606  this.genes[index] = ((value >>> 1 | bit << from) & mask) | (this.genes[index] & ~mask);
607 
608  return b;
609  } else {
610  // left shift
611  int submask_1 = from == 0 ? 0 : (0xFFFFFFFF >>> (bits - from));
612  int submask_2 = (to + 1) == bits ? 0 : (0xFFFFFFFF << (to + 1));
613 
614  int mask = ~(submask_1 ^ submask_2);
615  int value = this.genes[index] & mask;
616  int b = (value >> to) & 1;
617 
618  this.genes[index] = ((value << 1 | bit << from) & mask) | (this.genes[index] & ~mask);
619 
620  return b;
621  }

660  public final void swap(final int pos1, final int pos2) {
661 
662  int i1 = pos1 / load;
663  int offset1 = load - (pos1 % load) - 1;
664  int v1 = this.genes[i1];
665  int m1 = 1 << offset1;
666  int b1 = (this.genes[i1] >> offset1) & 1;
667 
668  int i2 = pos2 / load;
669  int offset2 = load - (pos2 % load) - 1;
670  int v2 = this.genes[i2];
671  int m2 = 1 << offset2;
672  int b2 = (this.genes[i2] >> offset2) & 1;
673 
674  if (i1 == i2) {
675  this.genes[i1] = (v1 & ~(m1 | m2)) | (b1 << offset2) | (b2 << offset1);
676  } else {
677  this.genes[i1] = (v1 & ~m1) | (b2 << offset1);
678  this.genes[i2] = (v2 & ~m2) | (b1 << offset2);
679  }
