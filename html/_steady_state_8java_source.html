171 
172  out.setAs(initial);
173 
174  int evolvedSize = evolved.size();
175  int outSize = out.size();
176 
177  int len = evolvedSize < outSize ? evolvedSize : outSize;
178  if (replacementRate < len) {
179  len = replacementRate;
180  }
181 
182  if (this.fitness != null) {
183  this.fitness.sort(evolved);
184  } else {
185  Fitness.sort(evolved, this.biggerIsBetter);
186  }
187 
188  if (this.replacementStrategy == ReplacementStrategy.WORST) {
189  out.sort();
190 
191  for (int i = 0; i < len; i++) {
192 
193  Individual<T> sibling = evolved.getIndividual(i);
194  Individual<T> candidate = out.getIndividual(outSize - i - 1);
195 
196  boolean better = this.fitness != null ? this.fitness.dominates(sibling, candidate) : Fitness.dominates(sibling, candidate, this.biggerIsBetter);
197 
198  if (this.elitist && !better) {
199  break;
200  }
201 
202  candidate.setAs(sibling);
203 
204  }
205  } else {
206  // RANDOM
207  for (int i = 0; i < len; i++) {
208 
209  Individual<T> sibling = evolved.getIndividual(i);
210 
211  for (int h = 0; h < 3; ++h) {
212  Individual<T> candidate = out.getIndividual(random.nextInt(outSize));
213  boolean better = this.fitness != null ? this.fitness.dominates(sibling, candidate) : Fitness.dominates(sibling, candidate, this.biggerIsBetter);
214 
215  if (this.elitist && !better) {
216  continue;
217  }
218 
219  candidate.setAs(sibling);

222  }
