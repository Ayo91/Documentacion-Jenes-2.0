39  public TravelSalesmanProblem(double[][] matrix, int popSize, int genLimit) {
40 
41  cities = matrix[0].length;
42 
43  IntegerChromosome chrom = new IntegerChromosome(cities, -1, cities - 1);
44  for (int i = 0; i < cities; ++i) {
45  chrom.setValue(i, i < cities - 1 ? i + 1 : 0);
46  }
47  Individual<IntegerChromosome> sample = new Individual<IntegerChromosome>(chrom);
48  Population<IntegerChromosome> pop = new Population<IntegerChromosome>(sample, popSize);
49 
50  algorithm = new TSPGA(matrix, pop, genLimit);
51 
52  AbstractStage<IntegerChromosome> selection = new TournamentSelector<IntegerChromosome>(2);
53  AbstractStage<IntegerChromosome> crossover = new TSPCrossover(1);
54  AbstractStage<IntegerChromosome> mutation = new TSPMutator(0.2);
55 
56  algorithm.addStage(selection);
57  algorithm.addStage(crossover);
58  algorithm.addStage(mutation);
59 
60  algorithm.setBiggerIsBetter(false);
61  }

85  private static double getDistance(int i, int j) {
