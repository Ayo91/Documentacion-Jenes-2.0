168  private Color[][] calcSignature(BufferedImage i, boolean showSignature) {
169  int n = this.nSamples;
170  // Get memory for the signature.
171  Color[][] sig = new Color[n][n];
172  // For each of the 25 signature values average the pixels around it.
173  // Note that the coordinate of the central pixel is in proportions.
174  float[] prop = new float[]{1f / 10f, 3f / 10f, 5f / 10f, 7f / 10f, 9f / 10f};
175  for (int x = 0; x < n; x++) {
176  for (int y = 0; y < n; y++) {
177  sig[x][y] = this.averageAround(i, prop[x], prop[y], showSignature);
178  }
179  }
180  return sig;
181  }

188  private Color averageAround(BufferedImage i, double px, double py, boolean showSignature) {
189 
190  // Get memory for a pixel and for the accumulator.
191  double[] pixel = new double[3];
192  double[] accum = new double[3];
193  // The size of the sampling area.
194  int sampleSize = this.sampleAreaSize;
195  int numPixels = 0;
196  // Sample the pixels.
197  Raster raster = i.getData();
198  for (double x = px * baseSize - sampleSize; x < px * baseSize + sampleSize; x++) {
199  for (double y = py * baseSize - sampleSize; y < py * baseSize + sampleSize; y++) {
200  raster.getPixel((int) x, (int) y, pixel);
201  accum[0] += pixel[0];
202  accum[1] += pixel[1];
203  accum[2] += pixel[2];
204 
205  numPixels++;
206  // only for debug mode... print on the surface of the image all the sample area painted with orange color
207  if (showSignature) {
208  i.setRGB((int) x, (int) y, 0xFF6600);
209  }
210  }
211  }
212 
213  // Average the accumulated values.
214  accum[0] /= numPixels;
215  accum[1] /= numPixels;
216  accum[2] /= numPixels;
217 
218  // only for debug mode... print on the image the areas evaluated as sample for distance evaluation
219  if (showSignature) {
220  for (double x = px * baseSize - sampleSize; x < px * baseSize + sampleSize; x++) {
221  for (double y = py * baseSize - sampleSize; y < py * baseSize + sampleSize; y++) {
222  i.setRGB((int) x, (int) y, new Color((int) accum[0], (int) accum[1], (int) accum[2]).getRGB());
223  }
224  }
225  }
226 
227  return new Color((int) accum[0], (int) accum[1], (int) accum[2]);
228  }

235  private double calcDistance(BufferedImage other) {
236  // Calculate the signature for that image.
237  Color[][] sigOther = calcSignature(other, false);
238  // There are several ways to calculate distances between two vectors,
239  // we will calculate the sum of the distances between the RGB values of
240  // pixels in the same positions.
241  double dist = 0;
242  int n = this.nSamples;
243  for (int x = 0; x < n; x++) {
244  for (int y = 0; y < n; y++) {
245  int r1 = signature[x][y].getRed();
246  int g1 = signature[x][y].getGreen();
247  int b1 = signature[x][y].getBlue();
248 
249  int r2 = sigOther[x][y].getRed();
250  int g2 = sigOther[x][y].getGreen();
251  int b2 = sigOther[x][y].getBlue();
252 
253  double tempDist = Math.sqrt(
254  (r1 - r2) * (r1 - r2)
255  + (g1 - g2) * (g1 - g2)
256  + (b1 - b2) * (b1 - b2));
257  dist += tempDist;
258  }
259  }
260  return dist;
261  }
