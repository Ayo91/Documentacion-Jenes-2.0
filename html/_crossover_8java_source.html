105  final long startTime=System.currentTimeMillis();
106  ((Statistics)this.statistics).crossovers=0;
107  out.setAs(in);
108  final int size=out.size();
109  for (int k=0; k<size-(spread-1); k+=spread ) { // -(spread-1) is necessary for avoiding to exceed
110  for (int i=0; i<spread; i++) {
111  offsprings[i]=out.getIndividual(k+i);
112  }
113  if (random.nextBoolean(this.probability)) {
114  cross(offsprings);
115  ((Statistics) this.statistics).crossovers++;
116  for (int i=0; i<spread; i++) {
117  offsprings[i].setNotEvaluated();
118  }
119  }
120  }
121  this.statistics.executionTime=System.currentTimeMillis()-startTime;
