547  @SuppressWarnings("unchecked")
548  public final void evolve(boolean restart) throws AlgorithmException {
549 
550  this.body.init(this);
551 
552  this.start(restart);
553 
554  final int limit = this.generationLimit;
555  for (generation = 0; generation < limit && !end(); generation++) {
556 
557  this.statistics.generations = generation + 1;
558  try {
559  currentPopulation = history[0];
560  nextPopulation = history[this.historySize - 1];
561  nextPopulation.setAs(currentPopulation);
562 
563  // Because Sequence uses a temporary population, it can be used
564  // nextPopulation
565  // for both input and output parameters, in this way
566  // memory is saved by avoiding the allocation of an additional
567  // population.
568  // (see Sequence.process)
569  this.body.process(nextPopulation, nextPopulation);
570 
571  this.evaluatePopulation(nextPopulation, fullEvaluationForced);
572 
573  for (int i = this.historySize - 1; i > 0; i--) {
574  history[i] = history[i - 1];
575  }
576  history[0] = nextPopulation;
577  } catch (StageException e) {
578  this.statistics.exceptionTerminated = true;
579  throw new AlgorithmException(
580  "An error occured during the ga evolution", e);
581  }
582  if (elitism > 0) {
583  this.applyElitism();
584  }
585  final long now = System.currentTimeMillis();
586  this.statistics.generationEndTimes[generation] = now;
587  this.onGeneration(now);
588  for (GenerationEventListener gel : generationListeners) {
589  gel.onGeneration(this, now);
590  }
591 
592  Pool<T> p = history[0].getPool();
593  if (p != null) {
594  p.resize();
595  }
596  }
597  this.stop();
598  }

643  ael.onAlgorithmStart(this, now);

775  public final void evaluatePopulation(final Population<T> population, final boolean forced) {
776 
777  if (this.runner != null) {
778  this.runner.onEvaluationBegin(population, forced);
779  }
780 
781  // notify to the population that
782  if (this.getFitness() != null) {
783  population.setEvaluatedBy(this.getFitness());
784  population.setSortingBy(this.getFitness().getBiggerIsBetter());
785  } else {
786  //this is done for backward compatibility
787  population.setEvaluatedBy(null);
788  population.setSortingBy(this.isBiggerBetter());
789  }
790 
791  long now = System.currentTimeMillis();
792  this.statistics.fitnessEvalStageBegin[this.generation] = now;
793 
794  for (Individual individual : population) {
795  if (!individual.isEvaluated() || forced || this.isFitnessChanged()) {
796  if (this.runner != null) {
797  this.runner.evaluateIndividual(individual);
798  } else {
799  this.evaluateIndividual(individual);
800  }
801 
802  this.statistics.fitnessEvaluationNumbers++;
803  }
804  }
805 
806  if (this.runner != null) {
807  this.runner.onEvaluationEnd();
808  }
809 
810  now = System.currentTimeMillis();
811  this.statistics.fitnessEvalStageEnd[this.generation] = now;
812  this.statistics.timeSpentInFitnessEval += now - statistics.fitnessEvalStageBegin[this.generation];
813 
814  }

1010  protected final void applyElitism() {
1011  int currentSize = currentPopulation.size();
1012  int nextSize = nextPopulation.size();
1013 
1014  int len = currentSize < nextSize ? currentSize : nextSize;
1015  if (elitism < len) {
1016  len = elitism;
1017  }
1018 
1019  Fitness fit = this.getFitness();
1020 
1021  if (fit != null) {
1022  fit.sort(currentPopulation);
1023  } else {
1024  Fitness.sort(currentPopulation, this.isBiggerBetter());
1025  }
1026 
1027  if (this.elitismStrategy == ElitismStrategy.WORST) {
1028  if (fit != null) {
1029  fit.sort(nextPopulation);
1030  } else {
1031  Fitness.sort(nextPopulation, this.isBiggerBetter());
1032  }
1033 
1034  for (int i = 0; i < len; i++) {
1035  nextPopulation.getIndividual(nextSize - i - 1).setAs(
1036  currentPopulation.getIndividual(i));
1037  }
1038  } else {
1039  // ElitismStrategy.RANDOM
1040  for (int i = 0; i < len; i++) {
1041  nextPopulation.getIndividual(random.nextInt(nextSize)).setAs(
1042  currentPopulation.getIndividual(i));
1043  }
1044  }
1045  }
