152  public final void process(Population<T> in, Population<T> out) throws StageException {
153  /*
154  * Distributes the initial population between its branchs and invokes their process method.
155  * At the end it merges all the branche output populations in the final parallel population
156  */
157  try {
158 
159  for (int i = 0; i < branchesIn.length; i++) {
160  this.branchesIn[i].clear();
161  }
162 
163  // distributes the inCopy population
164  this.distribute(in, branchesIn);
165 
166  // executes the stage on each branch
167  for (int i = 0; i < this.branchesIn.length; ++i) {
168  //set branchesOut population to the right size
169  this.branchesOut[i].resizeAs(branchesIn[i]);
170  this.stages.get(i).process(branchesIn[i], branchesOut[i]);
171  }
172 
173  // makes the final population from the branch subpopulations
174  this.mergePopulation(branchesOut, out);
175 
176  } catch (Exception e) {
177  throw new StageException(e.getMessage(), e);
178  }
179  }
