469  private <K extends Chromosome> List<Individual> sort(List<Individual<K>> in, final boolean[] bis) {
470 
471  this.list.clear();
472  this.list.addAll(in);
473 
474  if (this.mode == SortingMode.PARTIAL) {
475  if (bis[0]) {
476  Collections.sort(this.list, Collections.reverseOrder());
477  } else {
478  Collections.sort(this.list);
479  }
480  } else if (this.mode == SortingMode.HIERACHICAL) {
481  Collections.sort(this.list, new Comparator<Individual>() {
482 
483  public int compare(Individual i1, Individual i2) {
484 
485  double[] s1 = i1.getAllScores();
486  double[] s2 = i2.getAllScores();
487  int m = bis.length;
488 
489  if (s1.length != m || s2.length != m) {
490  throw new IllegalStateException("[Jenes]: individuals must have scores compatible with the number of objectives!");
491  }
492 
493  int i = 0;
494  DIFFERENT:
495  {
496  for (i = 0; i < m; ++i) {
497  if (s1[i] != s2[i]) {
498  break DIFFERENT;
499  }
500  }
501  return 0;
502  }
503  //minus servs to obtain the ascending order
504  return -(bis[i] && s1[i] > s2[i] || !bis[i] && s1[i] < s2[i] ? -1 : 1);
505  }
506  });
507  } else if (this.mode == SortingMode.DOMINANCE) {
508  Collections.sort(this.list, new Comparator<Individual>() {
509 
510  public int compare(Individual i1, Individual i2) {
511  //minus servs to obtain the ascending order
512  return -(Fitness.dominance(i1, i2, bis));
513  }
514  });
515 
516  } else { // this.mode == SortingMode.CROWDING
517  reset(list);
518 
519  sortByDominance(bis);
520  int r = assignRanks(bis);
521 
522  for (int k = 0; k < r; ++k) {
523  sortByDistance(segments[k], segments[k + 1], bis.length);
524  }
525 
526  list.clear();
527  for (CrowdingRef cr : refs) {
528  list.add(cr.ref);
529  }
530  }
531 
532  this.setIndividualsRank(list, bis);
533  return list;

689  private void updateDistance(int h, int from, int to) {
690 
691  for (int i = from; i < to; ++i) {
692  ordering[i] = i;
693  }
694 
695  //bouble sort...
696  for (int i = to; i > from; --i) {
697  for (int j = from; j < i - 1; ++j) {
698 
699  Individual<?> indj = refs[ordering[j]].ref;
700  Individual<?> indk = refs[ordering[j + 1]].ref;
701 
702  double fj = indj.getScore(h);
703  double fk = indk.getScore(h);
704 
705  if (fj < fk) {
706  int sw = ordering[j + 1];
707  ordering[j + 1] = ordering[j];
708  ordering[j] = sw;
709  }
710  }
711  }
712 
713  //retrieve max and min
714  double fmin = refs[ordering[to - 1]].ref.getScore(h);
715  double fmax = refs[ordering[from]].ref.getScore(h);
716 
717  /*
718  * We look for an individual (if any) that is already the lower-bound for the given goal,
719  * in order to avoid duplication of bounds with gaining infinite distance
720  */
721  if (h > 0 && !Double.isInfinite(refs[ordering[to - 1]].distance)) {
722  for (int i = to - 2; i >= 0 && refs[ ordering[i]].ref.getScore(h) == fmin; --i) {
723  if (Double.isInfinite(refs[ ordering[i]].distance)) {
724  int sw = ordering[i];
725  ordering[i] = ordering[to - 1];
726  ordering[to - 1] = sw;
727  }
728  }
729  }
730 
731  /*
732  * We look for an individual (if any) that is already the upper-bound for the given goal
733  */
734  if (h > 0 && !Double.isInfinite(refs[ordering[from]].distance)) {
735  for (int i = from + 1; i < ordering.length && refs[ ordering[i]].ref.getScore(h) == fmax; ++i) {
736  if (Double.isInfinite(refs[ ordering[i]].distance)) {
737  int sw = ordering[i];
738  ordering[i] = ordering[from];
739  ordering[from] = sw;
740  }
741  }
742  }
743 
744  refs[ordering[to - 1]].distance = Double.POSITIVE_INFINITY;
745  refs[ordering[from]].distance = Double.POSITIVE_INFINITY;
746 
747  for (int i = from + 1; i < to - 1; ++i) {
748  refs[ordering[i]].distance += fmax != fmin ? (refs[ordering[i - 1]].ref.getScore(h) - refs[ordering[i + 1]].ref.getScore(h)) / (fmax - fmin) : Double.POSITIVE_INFINITY;
749  }
750  }
