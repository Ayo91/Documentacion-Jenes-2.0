425  private double[] update(int[] data) {
426 
427  double avg = 0;
428  for (int i = 0; i < horizon; ++i) {
429  avg += data[i];
430  }
431  avg /= horizon;
432 
433  double sd = 0;
434  for (int i = 0; i < horizon; ++i) {
435  double z = data[i] - avg;
436  sd += z * z;
437  }
438  sd = Math.sqrt(sd / horizon);
439 
440  out[0] = avg;
441  out[1] = sd;
442
