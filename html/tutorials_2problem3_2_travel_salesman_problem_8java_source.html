86  public TravelSalesmanProblem(double[][] matrix) {
87 
88  cities = matrix[0].length;
89  map = matrix;
90 
91  IntegerChromosome chrom = new IntegerChromosome(cities,0,cities-1);
92  for( int i = 0; i < cities; ++i ) {
93  chrom.setValue(i, i < cities - 1 ? i+1 : 0);
94  }
95  Individual<IntegerChromosome> sample = new Individual<IntegerChromosome>(chrom);
96  Population<IntegerChromosome> pop = new Population<IntegerChromosome>(sample, POPULATION_SIZE);
97 
98  algorithm = new TSPGA(matrix, pop, GENERATION_LIMIT);

127  public void solvePC() {
128 
129  Individual<PermutationChromosome> sample = new Individual<PermutationChromosome>(new PermutationChromosome(cities));
130  Population<PermutationChromosome> pop = new Population<PermutationChromosome>(sample, POPULATION_SIZE);

132  Fitness<PermutationChromosome> fitness = new Fitness<PermutationChromosome>(false) {
133 
134  @Override
135  public void evaluate(Individual<PermutationChromosome> individual) {
136  PermutationChromosome chrom = individual.getChromosome();
137  double count = 0;
138  int size = chrom.length();
139  for(int i=0;i<size-1;i++){
140  int val1 = chrom.getElementAt(i);
141  int val2 = chrom.getElementAt(i+1);
142  count += TravelSalesmanProblem.this.map[val1][val2];
143  }
144  count += TravelSalesmanProblem.this.map[size-1][0];
145  individual.setScore(count);
146  }
147 
148  };

166  public static double[][] simpleMap( int cities ) {
167  double[][] matrix = new double[cities][cities];
168 
169  matrix[0][0] = 0;
170  for( int i = 1; i <= cities/2; ++i) {
171  matrix[0][i] = i;
172  matrix[0][cities-i] = i;
173  }
174 
175  for( int i = 1; i < cities; ++i ) {
176  for( int j = 0; j < cities; ++j ) {
177  matrix[i][(i+j)%cities] = matrix[0][j];
178  }
179  }
180  return matrix;
181  }

183  public static double[][] randomMap( int cities ) {
184  double[][] matrix = new double[cities][cities];
185  for( int i = 0; i < cities; ++i ) {
186  for( int j = 0; j < cities; ++j ) {
187  matrix[i][j] = i!=j ? Random.getInstance().nextDouble(MAX_DISTANCE) : 0;
188  }
189  }
